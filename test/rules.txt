Parser:
 - expr: add | neg
neg: ADD mul
add: (add ADD mul) | mul
mul: (mul MUL pow) | pow
pow: (pow POW atom) | atom
paren: LPAREN expr RPAREN
atom: NUM | ID | paren

// 2 * 4 + 5 ^ (4 + 3)
// atom MUL atom ADD atom POW atom LPAREN atom ADD atom RPAREN
// 


Lexer:
ID: `[a-zA-Z_][a-zA-Z0-9_]*`
NUM: `[0-9]+`
ADD: `\+|\-`
MUL: `\*|\/`
POW: `\^`
LPAREN: `\(`
RPAREN: `\)`
